## 1. Model 文件生成提示词

```
请根据以下信息生成 Sequelize Model 文件:

**实体名称**: [EntityName] (小写,如: employee, digital, flow)
**中文名称**: [ChineseName] (如: 员工, 数字人, 培训流程)
**字段定义**:
- [fieldName]: {类型, 注释, 默认值, 其他配置}
  例如:
  - name: {STRING, 名称}
  - status: {ENUM['open', 'closed'], 状态:正常,关闭, 默认值: 'open'}
  - options: {JSONB, 扩展字段, 默认值: {}}
  - createdAt/updatedAt: 系统自动添加,无需定义

**关联关系**:
- belongsTo: [关联表] (是否必填)
- hasMany: [关联表]
- 其他关联...

**索引配置**:
- 唯一索引: [字段组合]
- 普通索引: [字段]

**生成要求**:
1. 文件路径: server/libs/models/[entityName].js
2. 使用 DataTypes 定义字段类型
3. 需要关联租户(tenant)则使用 options.getTenantModels().tenant
4. 包含 comment、associate、options 配置
5. JSON/JSONB 字段设置合适的默认值
6. ENUM 类型定义枚举值
```

### Model 文件结构模板

```javascript
module.exports = ({ DataTypes, options }) => {
  return {
    model: {
      // 字段定义
      fieldName: {
        type: DataTypes.STRING,
        comment: '字段注释'
      },
      enumField: {
        type: DataTypes.ENUM('value1', 'value2'),
        defaultValue: 'value1',
        comment: '枚举字段'
      },
      jsonField: {
        type: DataTypes.JSON,
        defaultValue: {},
        comment: 'JSON字段'
      },
      jsonbField: {
        type: DataTypes.JSONB,
        comment: 'JSONB字段'
      }
    },
    associate: ({ modelName, relatedModel1, relatedModel2 }) => {
      // 关联关系定义
      modelName.belongsTo(options.getTenantModels().tenant, {
        allowNull: false
      });
      modelName.belongsTo(relatedModel1);
      modelName.hasMany(relatedModel2);
    },
    options: {
      comment: '模型注释',
      indexes: [
        {
          fields: ['fieldName1', 'fieldName2'],
          unique: true
        }
      ]
    }
  };
};
```

## 2. Service 文件生成提示词

```
请根据以下信息生成 Service 层文件:

**实体名称**: [EntityName] (小写)
**Model 引用**: models.[entityName]
**是否需要租户隔离**: [true/false]

**增删改查需求**:

### create - 创建
- 参数: [字段列表]
- 校验规则: [唯一性校验字段、必填校验等]
- 业务逻辑: [关联数据检查、数据预处理等]

### detail - 详情
- 参数: id
- 权限校验: [是否需要租户隔离校验]
- 异常处理: [资源不存在提示]

### save - 更新
- 参数: id, [可更新字段列表]
- 更新限制: [不允许更新的字段]
- 唯一性校验: [排除当前ID]

### remove - 删除
- 参数: id
- 关联处理: [需要级联删除的数据]
- 软删除/硬删除: [方式]

### list - 列表
- 参数: filter={筛选条件}, perPage, currentPage
- 筛选字段: [精确匹配字段]
- 模糊搜索: [支持关键字搜索的字段]
- 排序: [默认排序规则]
- 关联查询: [需要关联的表]
- 返回格式: {pageData: [], totalCount: 0}

### setStatus - 设置状态 (可选)
- 参数: id, status
- 状态枚举: [状态列表]
- 状态切换校验: [业务规则]

**辅助方法**:
- [其他业务方法名称及描述]

**生成要求**:
1. 文件路径: server/libs/services/[entityName].js
2. 使用 fastify-plugin 包装
3. 所有方法接收 authenticatePayload 参数进行权限校验
4. 使用 Sequelize 的 Op 操作符进行查询
5. 错误信息使用 throw new Error()
6. 使用 Object.assign 合并数据
7. 使用 lodash 的 omit/pick 等工具函数
8. 返回值格式统一
```

### Service 文件结构模板

```javascript
const fp = require('fastify-plugin');
const omit = require('lodash/omit');

module.exports = fp(async (fastify, options) => {
  const { models, services } = fastify[options.name];
  const tenantServices = fastify.tenant.services;
  const tenantModels = fastify.tenant.models;
  const { Op } = fastify.sequelize.Sequelize;

  // 创建
  const create = async (authenticatePayload, { name, ...data }) => {
    const { tenantId } = authenticatePayload;

    // 业务校验
    if (condition) {
      throw new Error('错误信息');
    }

    // 唯一性校验
    if (await models.entityName.count({ where: { uniqueField } }) > 0) {
      throw new Error('字段不能重复');
    }

    return await models.entityName.create(Object.assign({}, data, { tenantId }));
  };

  // 详情
  const detail = async (authenticatePayload, { id }) => {
    const { tenantId } = authenticatePayload;
    const entity = await models.entityName.findByPk(id);
    if (!entity) {
      throw new Error('未找到记录');
    }
    if (entity.tenantId !== tenantId) {
      throw new Error('未找到记录');
    }
    return entity;
  };

  // 更新
  const save = async (authenticatePayload, { id, ...data }) => {
    const entity = await detail(authenticatePayload, { id });
    const { tenantId } = authenticatePayload;

    // 唯一性校验(排除当前ID)
    if (
      await models.entityName.count({
        where: {
          uniqueField: data.uniqueField,
          tenantId,
          id: { [Op.not]: entity.id }
        }
      }) > 0
    ) {
      throw new Error('字段不能重复');
    }

    await entity.update(omit(data, ['tenantId', 'status']));
    return entity;
  };

  // 删除
  const remove = async (authenticatePayload, { id }) => {
    const entity = await detail(authenticatePayload, { id });

    // 级联删除关联数据
    await models.relatedModel.destroy({
      where: { entityId: entity.id }
    });

    await entity.destroy();
  };

  // 列表
  const list = async (authenticatePayload, { filter = {}, perPage, currentPage }) => {
    const { tenantId } = authenticatePayload;
    const whereQuery = {};

    // 精确匹配
    ['status', 'type'].forEach(name => {
      if (filter[name]) {
        whereQuery[name] = filter[name];
      }
    });

    // 模糊搜索
    if (filter['keyword']) {
      whereQuery[Op.or] = [
        { name: { [Op.like]: `%${filter['keyword']}%` } },
        { description: { [Op.like]: `%${filter['keyword']}%` } }
      ];
    }

    const { count, rows } = await models.entityName.findAndCountAll({
      include: tenantModels.org, // 关联查询
      where: Object.assign({}, whereQuery, { tenantId }),
      offset: perPage * (currentPage - 1),
      limit: perPage,
      order: [['createdAt', 'DESC']]
    });

    return {
      pageData: rows,
      totalCount: count
    };
  };

  // 设置状态
  const setStatus = async (authenticatePayload, { id, status }) => {
    const entity = await detail(authenticatePayload, { id });
    await entity.update({ status });
    return entity;
  };

  // 注册服务
  Object.assign(fastify[options.name].services, {
    entityName: { create, detail, save, remove, list, setStatus }
  });
});
```

```javascript
const fp = require('fastify-plugin');

module.exports = fp(async (fastify, options) => {
  const { services } = fastify[options.name];
  const { authenticate } = fastify.account;
  const { authenticate: tenantAuthenticate } = fastify.tenant;

  // 列表接口

    `${options.prefix}/on-boarding/entity-name/list`,
    {
      onRequest: [authenticate.user, tenantAuthenticate.tenantUser],
      schema: {
        summary: '实体列表',
        query: {
          type: 'object',
          properties: {
            filter: {
              type: 'object',
              default: {}
            },
            perPage: {
              type: 'number',
              default: 20
            },
            currentPage: {
              type: 'number',
              default: 1
            }
          }
        }

    async request => {
      return services.entityName.list(request.tenantUserInfo, request.query);
    }
  );

  // 创建接口
  fastify.post(
    `${options.prefix}/on-boarding/entity-name/create`,
    {
      onRequest: [authenticate.user, tenantAuthenticate.tenantUser],
      schema: {
        summary: '添加实体',
        body: {
          type: 'object',
          properties: {
            name: {
              type: 'string'
            },
            field1: {
              type: 'string',
              default: ''
            }
          },
          required: ['name']
        }
      }
    },
    async request => {
      return services.entityName.create(request.tenantUserInfo, request.body);
    }
  );

  // 更新接口
  fastify.post(
    `${options.prefix}/on-boarding/entity-name/save`,

      schema: {

              type: 'string'
```

### 4.2 Service (employee.js)

```javascript
const fp = require('fastify-plugin');
const omit = require('lodash/omit');

module.exports = fp(async (fastify, options) => {
  const { models, services } = fastify[options.name];
  const tenantServices = fastify.tenant.services;
  const tenantModels = fastify.tenant.models;
  const { Op } = fastify.sequelize.Sequelize;

  const create = async (authenticatePayload, { tenantOrgId, name, ...data }) => {
    const { tenantId } = authenticatePayload;
    if (tenantOrgId) {
      const org = await tenantServices.org.detail({ id: tenantOrgId });
      if (org.tenantId !== tenantId) {
        throw new Error('组织不存在');
      }
    }

    if (!(data.email || data.phone)) {
      throw new Error('手机号或邮箱不能同时为空');
    }

    if (data.email && (await models.employee.count({ where: { email: data.email.toLowerCase(), tenantId } })) > 0) {
      throw new Error('邮箱不能重复');
    }
    if (data.phone && (await models.employee.count({ where: { phone: data.phone, tenantId } })) > 0) {
      throw new Error('手机号不能重复');
    }

    return await models.employee.create(Object.assign({}, data, { tenantId: tenantId, tenantOrgId, name: String(name).toLowerCase() }));
  };

  const detail = async (authenticatePayload, { id }) => {
    const { tenantId } = authenticatePayload;
    const employee = await models.employee.findByPk(id);
    if (!employee) {
      throw new Error('未找到员工');
    }
    if (employee.tenantId !== tenantId) {
      throw new Error('未找到员工');
    }

    return employee;
  };

  const save = async (authenticatePayload, { id, tenantOrgId, name, ...data }) => {
    const employee = await detail(authenticatePayload, { id });
    const { tenantId } = authenticatePayload;
    if (tenantOrgId) {
      const org = await tenantServices.org.detail({ id: tenantOrgId });
      if (org.tenantId !== tenantId) {
        throw new Error('组织不存在');
      }
    }
    if (!(data.email || data.phone)) {
      throw new Error('手机号或邮箱不能同时为空');
    }

    if (
      data.email &&
      (await models.employee.count({
        where: {
          email: data.email.toLowerCase(),
          tenantId,
          id: { [Op.not]: employee.id }
        }
      })) > 0
    ) {
      throw new Error('邮箱不能重复');
    }
    if (
      data.phone &&
      (await models.employee.count({
        where: {
          phone: data.phone,
          tenantId,
          id: { [Op.not]: employee.id }
        }
      })) > 0
    ) {
      throw new Error('手机号不能重复');
    }
    await employee.update(Object.assign({}, omit(data, ['tenantId', 'status']), { tenantOrgId }, name && { name: String(name).toLowerCase() }));
    return employee;
  };

  const remove = async (authenticatePayload, { id }) => {
    const employee = await detail(authenticatePayload, { id });
    await models.flowData.destroy({
      where: { employeeId: employee.id }
    });
    await employee.destroy();
  };

  const setStatus = async (authenticatePayload, { id, status }) => {
    const employee = await detail(authenticatePayload, { id });
    await employee.update({ status });
    return employee;
  };

  const list = async (authenticatePayload, { filter = {}, perPage, currentPage }) => {
    const { tenantId } = authenticatePayload;
    const whereQuery = {};
    ['status'].forEach(name => {
      if (filter[name]) {
        whereQuery[name] = filter[name];
      }
    });
    if (filter['keyword']) {
      whereQuery[Op.or] = [
        {
          name: {
            [Op.like]: `%${filter['keyword']}%`
          }
        },
        {
          email: {
            [Op.like]: `%${filter['keyword']}%`
          }
        },
        {
          phone: {
            [Op.like]: `%${filter['keyword']}%`
          }
        }
      ];
    }

    const { count, rows } = await models.employee.findAndCountAll({
      include: tenantModels.org,
      where: Object.assign({}, whereQuery, {
        tenantId
      }),
      offset: perPage * (currentPage - 1),
      limit: perPage,
      order: [['createdAt', 'DESC']]
    });

    return {
      pageData: rows,
      totalCount: count
    };
  };

  Object.assign(fastify[options.name].services, {
    employee: { create, list, detail, save, remove, setStatus }
  });
});
```

## 5. 关键注意事项

### 5.1 命名规范

| 类型 | 命名规则 | 示例 |
|------|---------|------|
| 文件名 | 小写 + .js | employee.js, digital.js |
| Model 名称 | 首字母大写 | Employee, Digital |
| Service 方法 | 小驼峰 | create, detail, save, remove, list, setStatus |
| API 路径 | 小写 + 连字符 | /on-boarding/employee/list |

### 5.3 数据校验

```javascript
// 1. 必填字段校验
if (!name) {
  throw new Error('名称不能为空');
}

// 2. 唯一性校验 (使用 count + Op.not)
if (await models.entity.count({ where: { uniqueField } }) > 0) {
  throw new Error('字段不能重复');
}

// 3. 排除当前ID的唯一性校验 (更新时使用)
if (await models.entity.count({
  where: {
    uniqueField: data.uniqueField,
    id: { [Op.not]: entity.id }
  }
}) > 0) {
  throw new Error('字段不能重复');
}

// 4. 租户隔离校验
if (entity.tenantId !== tenantId) {
  throw new Error('无权操作此资源');
}
```
